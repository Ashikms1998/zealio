import { configureStore } from '@reduxjs/toolkit';
import { createSlice } from '@reduxjs/toolkit';

// Create a slice for the user data
const userSlice = createSlice({
  name: 'user',
  initialState: {
    name: "Ashik",
    age: "26"
  },
  reducers: {
    // You can add reducer functions here to update the state
    updateName: (state, action) => {
      state.name = action.payload;
    },
    updateAge: (state, action) => {
      state.age = action.payload;
    }
  }
});

// Export the actions
export const { updateName, updateAge } = userSlice.actions;

// Create and export the store
const store = configureStore({
  reducer: {
    user: userSlice.reducer
  },
  // The middleware option has been removed as it's not needed for this basic setup
});

export default store;
